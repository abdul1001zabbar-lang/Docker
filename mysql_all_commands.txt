================MYSQL with volume + network + compose====================
mkdir mysql-compose
cd mysql-compose
touch docker-compose.yml
vim docker-compose.yml
version:'3'
    services:
        mysql-db:
            image: mysql:8.0
            container_name:mysql-compose-c1
            environment:MYSQL_ROOT_PASSWORD
            ports:
                -"3306:3306"
            volumes:
                -mysql-data:/var/lib/mysql
            networks:
                -mysql-network   

        ubuntu:
            image:ubuntu
            container_name:ubuntu-c1
            tty:true
            networks:
                - myql-network
volumes:
  mysql-data:

networks:
  mysql-network:


docker compose up -d
docker logs mysql-compose-c1
#o/p :-ready for connections

docker exec -it ubuntu-c1 bash
apt update
apt install iputils-ping -y
ping mysql-db

apt install mysql-client -y
mysql -h mysql-db -u root -p
root123

================MYSQL named volume + network + compose ====================

mkdir mysql-compose-volume-network
cd mysql-compose-volume-network
touch docker-compose.yml
vim docker-compose.yml
version: '3'

services:

  mysql-db:

    image: mysql:8.0

    container_name: mysql-compose-c1

    environment:
      MYSQL_ROOT_PASSWORD: root123

    ports:
      - "3306:3306"

    volumes:
      - mysql-data:/var/lib/mysql

    networks:
      - mysql-network


  ubuntu:

    image: ubuntu

    container_name: ubuntu-c1

    tty: true

    networks:
      - mysql-network


volumes:

  mysql-data:


networks:

  mysql-network:


docker compose up -d
