    file.write("Hello Docker")

docker network create app-network

docker volume create app-data

mkdir docker-project

cd docker-project

cat > app.py <<EOF
from flask import Flask

app = Flask(__name__)

@app.route("/")
def home():
    file = open("/data/demo.txt","r")
    return file.read()

app.run(host="0.0.0.0", port=5000)
EOF

cat > requirements.txt <<EOF
flask
EOF

cat > Dockerfile <<EOF
FROM python:3.11

WORKDIR /app

COPY . .

RUN pip install -r requirements.txt

CMD ["python","app.py"]
EOF

docker build -t flask-volume-app .

docker run -d \
--name flask-c1 \
--network app-network \
-v app-data:/data \
-p 5000:5000 \
flask-volume-app

docker exec -it flask-c1 bash

echo "Hello From Shared Volume" > /data/demo.txt

exit

docker run -d \
--name nginx-c1 \
--network app-network \
-v app-data:/usr/share/nginx/html \
-p 8080:80 \
nginx
