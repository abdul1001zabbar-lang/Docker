==================Flask with named volume==================

mkdir flask-named-volume
cd flask-named-volume
touch app.py requirements.txt Dockerfile

docker build -t flasknamedvolume .
docker volume create flask-data
docker run -d --name flask-named-c1 -p 5000:5000 -v flask-data:/app flasknamedvolume

docker volume inspect flask-data

docker exec -it flask-named-c1 bash
echo "Hello docker>demo.txt
exit

#if deleted container data will store in volume, Now if again created container will see demo.txt

docker run -d --name flask-new-c1 -p 5000:5000 -v flask-data:/app flasknamedvolume
docker exec -it flask-named-c1 bash
ls
cat demo.txt

===============Flask with network=====================

cd flask-network
touch app.py requirements.txt Dockerfile

docker build -t flasknetwork .
docker network create flask-net
docker run -d --anem flask-c1 --network flask-net -p 5000:5000 flasknetwork

http://localhost:5000

docker run -it --name ubuntu-c1 --network flask-net ubuntu bash
